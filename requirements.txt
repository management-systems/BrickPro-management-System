# ==========================================
# 🛠️ SETUP INSTRUCTIONS
# ==========================================
#
# 1. Clone & Install:
#    cd BrickPro
#    npm install
#
# 2. Start Database (Docker):
#    cd infra && docker-compose up -d postgres redis
#
# 3. Backend:
#    cd backend
#    cp .env.example .env
#    npm install
#    npx prisma generate
#    npx prisma migrate dev
#    npm run dev                → runs on PORT 4000
#
# 4. Web App:
#    cd apps/web
#    npm install
#    npm run dev                → runs on PORT 3000
#
# 5. Mobile App:
#    cd apps/mobile
#    npm install
#    npx expo start            → runs on PORT 8081
#    # Press 'a' for Android, 'i' for iOS
#
# 6. Full Docker Setup (all services):
#    cd infra
#    docker-compose up          → all services start
#
